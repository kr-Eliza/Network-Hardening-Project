table inet filter {
	chain input {
		type filter hook input priority 0; policy drop;

		iif "lo" counter accept
		ct state established,related counter accept

		ip saddr 10.10.10.0/24 tcp dport 22 counter accept

		counter log prefix "NFT_IN_DENY " limit rate 10/minute
	}

	chain forward {
		type filter hook forward priority 0; policy drop;

		ct state established,related counter accept

		ip saddr 10.10.10.0/24 ip daddr 10.10.20.10 tcp dport 80 counter accept
		ip saddr 10.10.10.0/24 ip daddr 10.10.20.10 tcp dport 443 counter accept
		ip saddr 10.10.10.0/24 ip daddr 10.10.20.10 tcp dport 22 counter accept
		ip saddr 10.10.10.0/24 ip daddr 10.10.20.0/24 icmp type echo-request limit rate 5/second counter accept

		counter log prefix "NFT_FWD_DENY " limit rate 10/minute
	}

	chain output {
		type filter hook output priority 0; policy accept;
	}
}
